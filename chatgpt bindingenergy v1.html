import numpy as np # --- Semi-empirical mass formula (Rohlf-style coefficients; MeV) --- aV, aS, aC, aA, aP = 15.75, 17.8, 0.711, 23.7, 11.18 def pairing_term(Z, N): # + for even-even, - for odd-odd, 0 otherwise if (Z % 2 == 0) and (N % 2 == 0): return +1 if (Z % 2 == 1) and (N % 2 == 1): return -1 return 0 def B_total_SEMF(A, Z): N = A - Z if Z <= 0 or N <= 0: return np.nan delta = pairing_term(Z, N) * aP / (A**0.5) # common convention for total BE return ( aV*A - aS*(A**(2/3)) - aC*(Z*(Z-1))/(A**(1/3)) - aA*((A-2*Z)**2)/A + delta ) # For each A, pick the Z that maximises binding energy (valley of stability-ish) A_vals = np.arange(2, 241) Z_best = [] B_best = [] for A in A_vals: Z_candidates = np.arange(1, A) # ignore unphysical Z=0 or Z=A B_candidates = np.array([B_total_SEMF(A, Z) for Z in Z_candidates]) i = np.nanargmax(B_candidates) Zb = int(Z_candidates[i]) Bb = float(B_candidates[i]) Z_best.append(Zb) B_best.append(Bb) B_best = np.array(B_best) B_per_nucleon = B_best / A_vals # --- Interactive plot (Plotly) --- try: import plotly.graph_objects as go from plotly.subplots import make_subplots fig = make_subplots( rows=2, cols=1, shared_xaxes=True, subplot_titles=("Total binding energy B(A) (approx)", "Binding energy per nucleon B(A)/A (approx)") ) hover = [f"A={A}, Z≈{Z}, N≈{A-Z}
B≈{B:.1f} MeV
B/A≈{B/A:.3f} MeV" for A, Z, B in zip(A_vals, Z_best, B_best)] fig.add_trace(go.Scatter(x=A_vals, y=B_best, mode="lines", text=hover, hoverinfo="text"), row=1, col=1) fig.add_trace(go.Scatter(x=A_vals, y=B_per_nucleon, mode="lines", text=hover, hoverinfo="text"), row=2, col=1) fig.update_xaxes(title_text="Mass number A", row=2, col=1) fig.update_yaxes(title_text="B (MeV)", row=1, col=1) fig.update_yaxes(title_text="B/A (MeV)", row=2, col=1) fig.update_layout(height=700) fig.write_html("binding_energy_both.html", include_plotlyjs="cdn") print("Wrote: binding_energy_both.html (open it in a browser)") except ImportError: print("Plotly not installed. Install with: pip install plotly")
